# Representamos los intervalos cerrados [inicio, fin] usando listas
intervalo_A = [2, 8]
intervalo_B = [6, 9]

print("Intervalo A:", intervalo_A)
print("Intervalo B:", intervalo_B, "\n")


inicio_interseccion = max(intervalo_A[0], intervalo_B[0])  
fin_interseccion = min(intervalo_A[1], intervalo_B[1])    

if inicio_interseccion <= fin_interseccion:
    interseccion = [inicio_interseccion, fin_interseccion]
else:
    interseccion = None

print("La verdadera intersección del intervalo es:", interseccion)
